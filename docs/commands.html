Quick Start

Installation

# Install
go install github.com/Bharath-code/promptvault@latest

# Verify installation
promptvault --version

# Initialize with starter prompts
promptvault init

First Steps

# Open interactive TUI
promptvault

# Add your first prompt
promptvault add "React Hook Debugging" \
  --stack frontend/react/hooks \
  --content "Analyze this React component..." \
  --tags "debugging,react,hooks" \
  --models "claude-sonnet,gpt-4o"

# Search for prompts
promptvault search "useEffect"

# Export to Cursor
promptvault export --format cursorrules --stack frontend/react > .cursorrules

Managing Prompts

promptvault add
Add a new prompt 
# Basic usage
promptvault add "Prompt Title" --content "Prompt content here"

# With all options
promptvault add "React Component Review" \
  --stack frontend/react \
  --tags "review,quality,best-practices" \
  --models "claude-sonnet,gpt-4o" \
  --verified \
  --content "Review this React component..."

# Preview before adding
promptvault add "My Prompt" --content "..." --preview

# From stdin
cat prompt.txt | promptvault add "From File" --stack backend/python

# Auto-detect stack
cd my-react-project
promptvault add "Component" --content "..."
# → Auto-detects: frontend/react
Flag Shorthand Description
--content -c Prompt content
--stack -s Tech stack path
--tags — Comma-separated tags
--models — Model names
--verified — Mark as verified
--preview — Preview before adding
promptvault list
ls show list-all List all prompts 
# List all prompts
promptvault list
promptvault ls

# Filter by stack
promptvault list --stack frontend/react

# Short output
promptvault list --short

# JSON output
promptvault list --json | jq '.[] | .title'

# Verbose/Debug mode
promptvault list -v
promptvault list -vd
promptvault search
find query Full-text search 
# Basic search
promptvault search "useEffect dependencies"
promptvault find "react hooks"

# JSON output
promptvault search "api" --json | jq 'length'
promptvault create
Create with AI assistance NEW 
# Standard interactive
promptvault create

# AI-assisted (requires API key)
export ANTHROPIC_API_KEY=sk-ant-...
promptvault create --ai

Features: Variable detection, tag recommendations, stack auto-detection, anti-pattern warnings, quality scoring (0-100)

Testing & Quality

promptvault test
Test prompts NEW 
# Interactive test
promptvault test abc123

# Single test
promptvault test abc123 \
  --input "Convert to TypeScript" \
  --expected "TypeScript with types" \
  --model claude-sonnet

# View history
promptvault test abc123 --history
promptvault audit
Audit for decay NEW 
# Full audit
promptvault audit

# Critical only
promptvault audit --severity critical

# JSON for CI/CD
promptvault audit --json | jq '.issues_found'

# CI check
if [ $(promptvault audit --severity critical --json | jq '.issues_found') -gt 0 ]; then
  echo "❌ Critical issues found"
  exit 1
fi

Critical Deprecated models, low success rate
Warning Unused prompts (90+ days)
Info Old versions (180+ days)

promptvault diff
Compare versions NEW 
# Compare versions
promptvault diff abc123 1 2

# Compare with current
promptvault diff abc123 3 HEAD

# Compare last two
promptvault diff abc123 HEAD~1 HEAD
promptvault revert
Revert to version NEW 
# Revert to version 3
promptvault revert abc123 3

# With message
promptvault revert abc123 3 -m "Reverting broken changes"

Export & Integration

promptvault export
exp Export prompts 
# Export to SKILL.md
promptvault export --format skill.md --output SKILL.md

# Export specific stack
promptvault export --format cursorrules --stack frontend/react

# Export to AGENTS.md
promptvault export --format agents.md > AGENTS.md
Format Command Use Case
skill.md --format skill.md Claude Code skills
agents.md --format agents.md OpenAI Agents
cursorrules --format cursorrules Cursor IDE
windsurf --format windsurf Windsurf IDE
promptvault watch
Auto-export on changes NEW 
# Basic watch
promptvault watch --format skill.md --output SKILL.md

# Custom interval
promptvault watch --format cursorrules -o .cursorrules --interval 2s

# Filter by stack
promptvault watch -f skill.md -o REACT.md -s frontend/react

Sync & Backup

promptvault sync push
Backup to GitHub Gist 
# Backup to Gist
promptvault sync push --token <gh_token>

# Token from env
export PROMPTVAULT_GITHUB_TOKEN=ghp_...
promptvault sync push
promptvault sync pull
Restore from Gist 
# Restore from Gist
promptvault sync pull --token <gh_token>

Advanced Features

Global Flags

promptvault list -v

promptvault --help

JSON Output for Scripting

# Count prompts
promptvault list --json | jq 'length'

# Filter by stack
promptvault list --json | \
  jq '.[] | select(.stack | contains("react"))'

# Find unused
promptvault list --json | \
  jq '.[] | select(.usage_count == 0) | .title'

# Count by stack
promptvault list --json | \
  jq 'group_by(.stack) | map({stack: .[0].stack, count: length})'

Troubleshooting

Common Issues

"Command not found"

# Check installation
which promptvault

# Reinstall
go install github.com/Bharath-code/promptvault@latest

# Add to PATH
export PATH=$PATH:$(go env GOPATH)/bin

"Database locked"

# Close other instances
pkill -f promptvault

# Remove WAL files
rm ~/.promptvault/vault.db-wal
rm ~/.promptvault/vault.db-shm

"API key not set"

# Set API keys
export ANTHROPIC_API_KEY=sk-ant-...
export OPENAI_API_KEY=sk-...
